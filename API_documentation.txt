Server Side

Reset:

Login:
	Type: post 
	URL: /login
	Body: {
		user_name : <string>,
		password : <string>
	}
	Return: {
		success: <bool>,
		userid: <int>,
        message: <string>
	}

Sign Up:
	Type: post 
	URL: /signup
	Body: {
		user_name : <string>,
		password : <string>,
		email : <string>
	}
	Return: {
		success: <bool>,
		userid: <int>,
        message: <string>
	}

New group:
	Type: post 
	URL: /create_group
	Body: {
		user_id : <int>,
		group_name : <string>,
		group_info : <string>,
		members : <list<{
				participant_id: <int>,
				role: <string>
			}>>
	}
	Return: {
		success: <bool>,
		group_id : <int>,
		message: <string>
	}

Fetch Updates:
	Type: get
	URL: /get_updates
	Params: {
		last_opened: <time_stamp>,
		user_id: <int>
	}
	Return: {
		new_messages: <list<{
			sender_id: <int>,
			group_id: <int>,
			link: {
				link_id : <int>,
				title: <string>,
				link: <string>,
				desc: <string>,
				tags: <list<strings>>,
				timestamp: <time_stamp>
			}
		}>>,
		delete_messages: <list<{
			link_id: <int>
		}>>,
		react: <list<{
			user_id: <int>,
			link_id: <int>,
			react: <string>
		}>>,
		change_role: <list<{
			user_id: <int>,
			group_id: <int>,
			role: <string>
		}>>,
		remove_member: <list<{
			user_id: <int>,
			group_id: <int>
		}>>,
		add_user: <list<{
			user_id: <int>,
			group_id: <int>,
			user_name: <string>,
			role: <string>
		}>>,
		get_added: <list<{
			group_id: <int>,
			members: <list<{
				user_id: <int>,
				user_name: <string>,
				roles: <string>
			}>>
		}>>,
		timestamp: <time_stamp>
	}
